using System;

namespace first
{
    class Program
    {
        static void Main(string[] args)
        {
            int result = Addresult(23, 900);
            Console.WriteLine("結果"+result);
        }

        // Methodメソッドを作り、引数を宣言
        static int Addresult(int num, int room)
        {
            Console.WriteLine("変更前のnum=" + num);
            num += room;
            Console.WriteLine("変更後のnum=" + num);
            return num;
        }
    }
}
